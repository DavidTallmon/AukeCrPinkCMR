######## Model Parameter Presentations #####
#### Table of back transformed parameter estimates
mods <- list(
  "Length + Temp" = Egg.LeT,
  "Arrival + Length + Temp" = Egg.ALeT,
  "Length + Density + Temp" = Egg.LeDT
)

coef_table <- map_dfr(
  names(mods),
  function(m) {
    tidy(mods[[m]],
         effects = "fixed",
         component = "cond",
         conf.int = TRUE) %>%
      mutate(
        Model = m,
        Estimate_BT = exp(estimate),
        Lower_BT = exp(conf.low),
        Upper_BT = exp(conf.high)
      ) %>%
      dplyr::select(
        Model,
        term,
        Estimate_BT,
        Lower_BT,
        Upper_BT,
        p.value
      )
  }
)

wide_table <- coef_table %>%
  mutate(
    Effect = sprintf(
      "%.3f (%.3f, %.3f)",
      Estimate_BT,
      Lower_BT,
      Upper_BT
    )
  ) %>%
  dplyr::select(Model, term, Effect) %>%
  pivot_wider(
    names_from = Model,
    values_from = Effect
  )
print(wide_table)

##modified below from what was found here https://cran.r-project.org/web/packages/glmmTMB/vignettes/model_evaluation.pdf
##exponentiated version, reset to 1 baseline instead of 0

tidy_irr <- bind_rows(
  tidy(Egg.LeT, effects = "fixed", component = "cond", exponentiate = TRUE, conf.int = TRUE) %>%
    mutate(Model = "Best Model"),

  tidy(Egg.ALeT, effects = "fixed", component = "cond", exponentiate = TRUE, conf.int = TRUE) %>%
    mutate(Model = "Second Best Model"),

  tidy(Egg.LeDT, effects = "fixed", component = "cond", exponentiate = TRUE, conf.int = TRUE) %>%
    mutate(Model = "Third Best Model")
) %>%
    filter(term != "(Intercept)") %>%
    rename(
      Term = term,
      Estimate = estimate
    ) %>%
  mutate(Variable = forcats::fct_rev(factor(Term)))

ggplot(tidy_irr,
       aes(x = Estimate,
           y = Variable,
           xmin = conf.low,
           xmax = conf.high)) +
  geom_vline(xintercept = 1,
             linetype = "dashed",
             color = "grey50") +
  geom_errorbarh(width = 0.2) +
  geom_point(size = 2) +
  facet_grid(Model ~ ., scales = "free_y", space = "free_y") +
  scale_x_continuous(
      breaks = function(x) sort(unique(c(pretty(x), 1))),
      labels = scales::label_number(accuracy = 0.01)
  ) +
  labs(
    title = "Conditional Fixed Effects",
    x = "Multiplier Effect Size",
    y = "Variable"
  ) +
  theme_minimal() +
  theme(
    legend.position = "none",
    strip.text.y = element_text(face = "bold"),
    strip.background = element_rect(fill = "grey95", color = NA)
  )
